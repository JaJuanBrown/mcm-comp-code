get_tricky_letter_count <- function(phrase) {
  counter <- 0
  for (i in unlist(strsplit(phrase, ""))) {
    if ( i %in% c("r", "u", "w", "z", "x", "y")) {
      counter <- counter + 1 
    }   
  }
  counter
